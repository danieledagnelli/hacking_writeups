def index(request):

def user_logout(request):
    logout(request)
    return redirect(reverse('auth_login'))



def account_register(request):
    if request.method == 'POST':
        form = CustomUserCreationForm(request.POST)
        if form.is_valid():
            # process the form data
            form.save()
            return redirect(reverse('auth_login'))
        else:
            # form is not valid
            context = {"title": "Account Register", "form": form}
    else:
        form = CustomUserCreationForm()
        context = {"title": "Account Register", "form": form}
    return render(request, 'register.html', context)



def auth_login(request):
    if request.method == 'POST':
        # check creds
        form = AuthenticationForm(data=request.POST)
        if form.is_valid():
            username = form.cleaned_data.get('username')
            password = form.cleaned_data.get('password')
            user = authenticate(request, username=username, password=password)
            if user is not None:
                login(request, user)
                log_login_attempt(request, username, 'SUCCESSFUL')
                if 'next' in request.GET:
                    return HttpResponseRedirect(request.GET["next"])
                else:
                    return redirect(reverse('profile'))
        else:
            log_login_attempt(request, request.POST["username"], 'FAILED')
    else:
        form = AuthenticationForm()
    return render(request, 'login.html', {'form': form, 'title': 'Login'})



@login_required
def profile(request):
    user = request.user
    if request.method == 'POST':
        form = CustomUserProfileForm(request.POST, instance=user)
        if form.is_valid():
            form.save()
            return redirect('profile')
    else:
        form = CustomUserProfileForm(instance=user)
    context = {
        'form': form,
    }
    return render(request, 'profile.html', context)



@protected_resource(scopes=['read'])

    response_data = {
